[description]
Pipeline startup script - executed very early during asset compilation to configure diagnostic warning levels and import Configuration.lua.

[version]
2011-2013 (Battlefield 3, Medal Of Honor, Army Of Two, NFS The Run)
[notes]
Original pipeline startup with verification tag system.

Features:
- Environment.UserName from USERNAME environment variable
- BRANCH_NAME environment variable validation (required)
- Verification tag system: expectVerificationTag = '22062007'
- verificationTag runtime check for script/binary compatibility
- allScriptsRoot = '/Data/Scripts/'
- scriptsRoot = allScriptsRoot .. 'Pipeline/'
- core.import(scriptsRoot .. 'Configuration.lua')

Verification Tag Purpose:
Ensures the Lua script version matches the compiled C++ runtime version. When Frostbite's internal APIs changed, both the script and binary needed updates. Mismatches would cause crashes or incorrect behavior.

Why it was simple: Early Battlefield games had straightforward pipeline needs without complex diagnostic systems.
[content]
-- Pipeline startup script
--
--   This is executed very early, and can really only be used to
--   configure settings.
--
--   The settings specified here will override any settings in the
--   legacy settings file pipeline.config
--

core = require "Frost.Core"

require "os"
require "vfs"

Environment = {}

local userName = os.getenv("USERNAME")

if userName then Environment.UserName = userName end

branchName = os.getenv("BRANCH_NAME")

if not branchName then 
	error("Configuration error: BRANCH_NAME environment variable not set -- fix the .diceconfig file for this branch!") 
end

-- This is a verification tag to make sure the script is in sync with runtime
-- whenever something is changed that requires changes to be present on both
-- the script end and the runtime end, this should be updated. The value is
-- just an arbitrary string and has no semantic relevance

local expectVerificationTag = "22062007"

if verificationTag ~= expectVerificationTag then
	error("verification tag mismatch! Most likely, your data/Scripts folder is not right for your build!")
end

-- Execute configuration and setup scripts

allScriptsRoot = "/Data/Scripts/"
scriptsRoot = allScriptsRoot .. "Pipeline/"

core.import(scriptsRoot .. "Configuration.lua")

[version]
2022 (Need For Speed Unbound, Dead Space)
[notes]
Time-based diagnostic levels introduced.

Major New Features:
- make_level_timespan(level) - Time-limited diagnostic levels
- level_1_timespan through level_4_timespan
- get_level() enhanced with fallback: Pipeline.HintThreshold + 1

Time-based Diagnostic System:
Allows temporary warning escalation for specific date ranges:
- level_2_timespan('newFeatureWarning', '01/15/2023', 30)
- Sets warning to level 2 for 30 days starting January 15, 2023
- Maximum 60-day window enforced

Date Parsing:
- Format: MM/DD/YYYY
- Uses os.time() and os.difftime() for date arithmetic
- Calculates days since start date

Why This Matters:
When rolling out new pipeline features, teams needed warnings to be loud initially (to catch issues) but automatically quiet down after the transition period. This prevents warning fatigue while ensuring new issues are caught.

Technical:
- daysSinceStart = os.difftime(os.time(), startDate) / 86400
- Converts time difference to days (86400 seconds per day)
- Only applies level if within [0, numberOfDays] window
[content]
-- Pipeline startup script
--
--   This is executed very early, and can really only be used to
--   configure settings.
--
--

core = require "Frost.Core"

require "os"
require "vfs"

Environment = {}

local userName = os.getenv("USERNAME")

if userName then Environment.UserName = userName end

branchName = os.getenv("BRANCH_NAME")

if not branchName then 
	error("Configuration error: BRANCH_NAME environment variable not set -- fix the .diceconfig file for this branch!") 
end

-- Define functions used for configuration of warnings and other diagnostics

diag_info = {}

function make_level(level)
	return function(identifier)
		diag_info[identifier] = level
	end
end

function make_level_timespan(level)
	return function(identifier, dateStr, numberOfDays)
		local maxDays = 60
		if numberOfDays > maxDays then
			error(string.format("make_level_timespan cannot be set for more than %s days in the future", maxDays))
			return
		end
	
		local month, day, year = dateStr:match('^(%d%d)/(%d%d)/(%d%d%d%d)$')
		local startDate = os.time{month = tonumber(month), day = tonumber(day), year = tonumber(year)}

		local daysSinceStart = os.difftime(os.time(), startDate) / 86400 -- seconds between days / seconds per day

		if daysSinceStart > 0 and daysSinceStart < numberOfDays then
			local setLevel = make_level(level)
			setLevel(identifier)
		end
	end
end

function get_level(identifier)
	local level = diag_info[identifier] -- Returns nul or warning level
	if level == nil then
		return Pipeline.HintThreshold + 1
	end

	return level
end

level_1 = make_level(1)
level_2 = make_level(2)
level_3 = make_level(3)
level_4 = make_level(4)

level_1_timespan = make_level_timespan(1)
level_2_timespan = make_level_timespan(2)
level_3_timespan = make_level_timespan(3)
level_4_timespan = make_level_timespan(4)

-- Execute configuration and setup scripts

allScriptsRoot = "/Data/Scripts/"
scriptsRoot = allScriptsRoot .. "Pipeline/"

core.import(scriptsRoot .. "Configuration.lua")

--
