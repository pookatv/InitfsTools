[description]
Debug and testing configuration - controls thin-client mode, automated testing, and development settings (note that applySettings list is reference only; not accurate per game).

[version]
2011-2015 (Battlefield, Battlefront, Garden Warfare, Need For Speed)
[notes]
Original thin-client and testing implementation.

Features:
- Thin-client mode: Single mode (0=Off, 1=Console)
- Testing mode: Binary on/off
- Helper functions: getSetting(), setSetting()
- applySettings() wrapper around core.parseKeyValueCfgString()

Thin-client disables rendering, audio, input, and most client systems for headless server testing.
Testing mode enables monkey testing layers and debug entities.
[content]
core = require "Frost.Core"
logging = require "Frost.Logging"

local function applySettings(settings)
	core.parseKeyValueCfgString(settings, _G) end
	
-- helpers

local function getSetting(key)
	container, field = string.match(key, "(%w+)%.(%w+)");
	if container and field then
		if not _G[container] then
			_G[container] = {}
		end
		return _G[container][field]
	else
		return _G[key]
	end
end

local function setSetting(key, value)
	container, field = string.match(key, "(%w+)%.(%w+)");
	if container and field then
		if not _G[container] then
			_G[container] = {}
		end
		_G[container][field] = value
	else
		_G[key] = value
	end
end

-- thin-client settings: 0=Off 1=Console

setSetting("thin-client", cmdLineOption("thinClient", getSetting("thin-client")))

if getSetting("thin-client") == "1" then
	applySettings([=[
		RenderDevice.NullDriverEnable true
		RenderDevice.CreateMinimalWindow true
		Core.HardwareProfile Hardware_Low
		Core.HardwareGpuBias -1

		Client.DebrisClusterEnabled false

		Client.WorldRenderEnabled false
		Client.EmittersEnabled false

		Client.OvergrowthEnabled false
		Client.OccludersEnabled true
		Decal.SystemEnable false
		Render.DebugRendererEnable false
		UI.HudEnable false
		UI.System UISystem_None		
		EmitterSystem.Enable false
		Window.Fullscreen false
		Render.Enable false
		disable-input true
		scripted-input true
		use-script-input true
		disable-client-correction true
		disable-client-animations true
		Sound.Enable false
