[description]
Core utility module - provides file I/O, config parsing, and script importing

[version]
2011-2012 (Battlefield 3, Need For Speed The Run, Army Of Two)
[notes]
Original implementation with basic config file parsing.

Features:
- import() - Execute scripts with error handling
- readKeyValueCfg() - Parse old-style config files
- parseKeyValueCfgLine() - Handle individual config lines (supports ONE level of dotting: container.field)
- parseKeyValueCfgString() - Parse config strings

Comment removal: Strips both # and // style comments

Why it was simple: Early Battlefield games had simple config needs.

require "os"
require "vfs"

module("Frost.Core", package.seeall);

-- Execute a script
--
--   If flags contains a "silent=true" flag, then no message
--   will be emitted if the script can't be found
--

-- Execute a script, optionally returning an error message if it fails

function import(script, friendlyName, flags)
	local name = friendlyName or script

	local success, err = pcall(dofile,script)
	
	if not success and not (flags and flags.silent) then
		error("Could not import script: " .. name .. " error: " .. err)
	end
end

-- Execute a new-style config file

local function executeCfg(file)
	if log then
		log:debug("  Executing new style (lua) config file")
	end

	local script = file:read("*all")

	local func, err = loadstring(script)

	if func then 
		func() 
	else 
		error(err) 
	end
end

-- Parse a single line containing a key-value pair (and possibly some comments)

function parseKeyValueCfgLine(line, targetTable)
	-- Remove comments
	line = string.gsub(line, "#.*", ")
	line = string.gsub(line, "//.*", ")
	
	-- Split into key, value pairs
	local key, value = string.match(line, "(%S+)%s+(.+)%s*");
	
	if key and value then
		-- Handle dotted key names
		local container, field = string.match(key, "(%w+)%.(%w+)");
		
		if container and field then
			if not targetTable[container] then
				targetTable[container] = {}
			end
			
			targetTable[container][field] = value;

			if log then
				log:debug(string.format("set %s.%s = %s (from key-value cfg line)", container, field, value))
			end
		else
			targetTable[key] = value;

			if log then
				log:debug(string.format("set %s = %s (from key-value cfg line)", key, value))
			end	
		end
	end
end		

-- Parse a string of key-value config entries

function parseKeyValueCfgString(str, targetTable)
	for line in string.gmatch(str, "[^\n]*") do
		parseKeyValueCfgLine(line, targetTable)
	end
end

-- Parse an old-style key-value CFG file

function readKeyValueCfg(file, targetTable)
	if log then
		log:debug("Reading cfg '" .. file .. "'")
	end

	local f = vfs.open(file, "r")
	
	if not f then
		-- File not found
		if log then
			log:debug("Failed to find file '" .. file .. "'")
		end
			
		return
	end

	local firstLine = true

	while true do
		local line = f:read("*line")
		
		if not line then 
			break 
		end

		if firstLine and line == "#!/bin/lua" then
			executeCfg(f)
		end
		
		parseKeyValueCfgLine(line, targetTable)
	end
	
	f:close();
end

-- Parse the value of a key-value setting.  May call itself recursively to handle parsing of structs.

local function parseKeyValueCfgValue(text)
	-- TODO: implement context-free grammer so we can match structs of structs or structs of arrays
	-- For now, only structs of primitive types are implemented
	-- Arrays are not implemented

	local innerText = string.match(text, "^%s*{(%s*.-%s*)}%s*$") -- Match { <blah blah text> }
	if innerText then
		local returnValue = {}
		local elementCount = 0
		for assignment in string.gmatch(innerText, "[^,]+") do		-- Match sections delimited by commas
			local assignmentKey,assignmentValue = string.match(assignment, "^%s*(%S+)%s*=%s*(.-)%s*$")  -- Match "key = value" (without the quotes)
			if assignmentKey and assignmentValue then
				elementCount = elementCount + 1
				returnValue[assignmentKey] = assignmentValue
			end
		end
		-- Cull empty tables; the game asserts when parsing structs if it encounters {}		
		if elementCount == 0 then
			return "
		end
		
		return returnValue
	else
		return text
	end
end


-- Parse a single line containing a setting (and possibly some comments)

function parseKeyValueCfgLine(line, targetTable)
	local command -- The command to execute; default is "set", can be set to other values with .<commandname> syntax,
	local fullFieldName -- The full fieldname, either <singlefield> or <category>.<leaffield>
	local leafFieldName -- The last field in fullFieldName, the key that will be set in the parentContainer table
	local value -- The value being set (e.g. "dooby" in the line Core.Data dooby)
	local parentContainer -- The table that will be modified; parentContainer[leafFieldName] is the value being set
	local text -- The remaining text of the line to parse; this function has several stages that each chop off text from the line
	local restOfLine -- A temporary variable used in the regexes to assign the rest of the line
	
	text = line
	
	-- Remove comments
	text = string.gsub(text, "#.*", ")
	text = string.gsub(text, "//.*", ")
		
	-- recognize special commands identified by a leading . (e.g. .append for arrays)
	command, restOfLine = string.match(text, "^%s*%.(%S+)%s+(..-)%s*$")
	if command and restOfLine then
		command = command:lower()
		text = restOfLine
		
		-- exit if the command is not recognized
		if command == "set" then
			-- set command, same as the default syntax without a command
		elseif command == "append" then
			-- append work is done at the end of the function
		else
			return
		end
	else
		command = "set"
	end
			
	-- Split into key, value pairs
	fullFieldName, restOfLine = string.match(text, "^%s*(%S+)%s+(..-)%s*$");
	if not fullFieldName or not restOfLine then
		-- check for setting equal to empty string
		fullFieldName = string.match(text, "^%s*(%S+)%s*$")
		if not fullFieldName then
			return
		end
		restOfLine = "
	end
	text = restOfLine
		
	-- parse the value
	value = parseKeyValueCfgValue(text)
	if not value then
		return
	end
	
	-- Handle dotted key names
	parentContainer = targetTable
	leafFieldName = fullFieldName
	repeat
		local remainingFieldName
		local containerName
		
		containerName, remainingFieldName = string.match(leafFieldName, "^([^%.]+)%.(.+)$");
		if containerName then
			if not parentContainer[containerName] then
				parentContainer[containerName] = {}
			end
			parentContainer = parentContainer[containerName]
			leafFieldName = remainingFieldName
		end
	until not containerName

	-- Take action based on command, e.g. merely set the value, or append it to an array
	if command == "set" then
		parentContainer[leafFieldName] = value
	elseif command == "append" then
		if not parentContainer[leafFieldName] then
			parentContainer[leafFieldName] = {}
		elseif type(parentContainer[leafFieldName]) ~= "table" then
			return
		end
		parentContainer[leafFieldName][#parentContainer[leafFieldName] + 1] = value
	end

	if log then
		log:debug(string.format("%s %s = %s (from key-value cfg line)", command, fullFieldName, text))
	end	
end

-- These are functions to do unit testing of parseKeyValueCfgString
-- mrp: I copied the functions and test code to a separate file and authored it there, calling testparseKeyValueCfgLine from the lua interpreter command line
local function varequals(a,b)
	if type(a) ~= type(b) then
		return false
	end
	
	if type(a) == "table" then			
		local lengthA = 0
		local lengthB = 0
		local aIsArray = false
		local bIsArray = false
		
		for k,v in pairs(a) do
			lengthA = lengthA + 1
		end
		for k,v in pairs(b) do
			lengthB = lengthB + 1
		end
		aIsArray = lengthA == #a
		bIsArray = lengthB == #b
	
		if lengthA ~= lengthB then
			return false
		end
		if aIsArray ~= bIsArray then
			return false
		end
		if aIsArray then
			for i = 1, lengthA do
				if not varequals(a[i],b[i]) then
					return false
				end
			end
		else
			for k,v in pairs(a) do
				if not varequals(b[k],v) then
					return false
				end
			end
		end
		return true
	else
		return a == b
	end
end
	
local function vartostring(var)
	if not var then
		return "nil"
	end
	
	if type(var) == "table" then
		local isArray,numElements,text,first
		
		numElements = 0
		for k,v in pairs(var) do
			numElements = numElements + 1
		end
		if numElements == 0 then
			return "{ }"
		end
		isArray = #var == numElements
		
		text = "{ "
		first = true
		for k,v in pairs(var) do
			numElements = numElements + 1
			if first then
				first = false
			else
				text = text .. ", "
			end
			if isArray then
				text = text .. vartostring(v)
			else
				text = text .. string.format("%s: %s", vartostring(k), vartostring(v))
			end
		end
		if numElements == 0 then
			return "{ }"
		else
			text = text .. " }"
			return text
		end
	else
		return tostring(var)
	end
end
local function assertResults(text, expected)
	local testTable = {}
	for line in string.gmatch(text, "[^\n]+") do
		parseKeyValueCfgLine(line, testTable)
	end
	if not varequals(testTable,expected) then
		print("Expected: " .. vartostring(expected))
		print("Actual:   " .. vartostring(testTable))
		assert(false)
	end
end	
local function testparseKeyValueCfgLine()
	-- Handles key value
	assertResults("alpha beta", { alpha = "beta" })
	
	-- Handles category.key value
	assertResults("alpha.beta gamma", { alpha = { beta = "gamma" } })
	
	-- Handles variable names with arbitrary number of dots
	assertResults("alpha.beta.gamma.delta epsilon", { alpha = { beta = { gamma = { delta = "epsilon" } } } })
	
	-- Handles key with an empty value
	assertResults("alpha", { alpha = " })
	
	-- Handles whitespace everywhere (except around the . separating category from field name)
	assertResults(" alpha.beta gamma", { alpha = { beta = "gamma" } })
	
	-- Handles comments
	assertResults([[
# alpha beta
// gamma delta
	# epsilon omicron?
	// ran outofgreek
	anyway This shows up
	]], { anyway = "This shows up" })

	-- Handles .append
	assertResults([[
.append alpha.beta gamma
.append alpha.beta delta
	.append alpha.beta whitespace is handled
	.append another array
	]], { alpha = { beta = { "gamma", "delta", "whitespace is handled" } }, another = { "array" } })
	
	-- Append is ignored if appending to a non-table
	assertResults([[
	overwrites ignored
	.append overwrites deadbeef
	]], { overwrites = "ignored" })
	
	-- Invalid commands are ignored
	assertResults(".invalid alpha beta", {})
	
	-- .set works as default
	assertResults(".set alpha beta", { alpha = "beta" })
	
	-- empty tables in the value are culled (the game engine asserts if encounters {} when parsing a struct)
	assertResults([[
	alpha {}
	]], { alpha = " })
	
	-- Handles structs
	assertResults("alpha { a = vala, b = valb }", {alpha = {a = "vala", b = "valb"} })
	
	-- Handles arrays of structs
	assertResults([[
	.append alpha { a = vala, b = valb }
	.append alpha { a = vala2, b = valb2 }
	]], {alpha = {{a ="vala", b = "valb"},{a="vala2", b="valb2"}} })
	
	-- historical interest, the very first use of .append - an array of LogChannelSettings
	assertResults(".append Core.ChannelSettings {Channel=CSM, WarningResponse=CllTelemetry, InfoResponse=CllInfo}",
		{ Core = { ChannelSettings = { { Channel = "CSM", WarningResponse = "CllTelemetry", InfoResponse = "CllInfo" } } } })
		
	print("All tests succeeded.")

function parseKeyValueCfgLine(line, targetTable)
	-- Remove comments

	line = string.gsub(line, "#.*", ")
	line = string.gsub(line, "://", ":\\/\\/") -- Replace :// with :\/\/, e.g.: https://
	line = string.gsub(line, "//.*", ")

	-- Split into key, value pairs
	local key, value = string.match(line, "(%S+)%s+(.+)%s*")

	if key and value then
		-- Handle dotted key names

		local currentTable = targetTable
		local first, second
		second = key
		repeat
			first, second = string.match(second, "([^%.]+)%.?(.*)")

			if first and second then
				if second:len() == 0 then
					currentTable[first] = trim(value)
					if log then
						log:debug(string.format("set %s = %s (from key-value cfg line)", key ,value))
					end

					if not currentTable[first] then
						currentTable[first] = {}
					end
					currentTable = currentTable[first]
				end
			end

-- Parses the commandline settings and sets them in a global table for easy access
function parseCmdLineSettings()
	if allowCommandlineSettings then
		for _,option in ipairs(commandLine) do
			local table,member = string.match(option, "^%-(%w+)%.([%w%.]+)")
	
			if table and member then
				local value = commandLine[_ + 1] 
				local cfgLine = table .. "." .. member .. " " .. value
				core.parseKeyValueCfgString("Core.lua#commandLine", cfgLine, _G)
			else
				local varName = string.match(option, "^%-([%w-]+)")
				if varName then
					local value = commandLine[_ + 1] 
					if value then
						_G[varName] = value
					end
				end
			end
		end
	end

-- Helper functions to get/set cmdline arg values
function getSetting(key)

	if container and field then
		if not _G[container] then
			_G[container] = {}
		end
		val = _G[container][field]
	else
		val = _G[key]

    for key,value in pairs(a) do

require "os"
require "vfs"

module("Frost.Core", package.seeall);

-- Execute a script
--
--   Optional flags:
--		required (default: true):
--			Emits an error if the script doesn't exist.
--		silent (default: false):
--			DEPRECATED: Previous behaviour did not emit errors if the script did not exist OR failed to execute correctly.
--			`silent = true` has the same behaviour as `required = false` now, but the name is misleading.
--

-- Execute a script, optionally returning an error message if it fails
function import(script, friendlyName, flags)
	local name = friendlyName or script
	
	if not flags then
		flags = {}
	end
	if flags.required == nil and not flags.silent then
		flags.required = true
	end

	local success, executeError = pcall(dofile,script)

	if not success then
		if string.match(executeError, "file not found") then
			if flags.required then
				error("Failed to open file: " .. name)
			elseif log then
				log:debug("Script not executed; failed to open file: " .. name)
			end
		else
			error("Failed to execute script: " .. name .. " error: " .. executeError)
		end
	end
end

-- Execute a new-style config file
local function executeCfg(file)
	if log then
		log:debug("  Executing new style (lua) config file")
	end

	local script = file:read("*all")

	local func, err = loadstring(script)

	if func then 
		func() 
	else 
		error(err) 
	end
end

function trim(s)
  return s:match'^%s*(.*%S)' or ''
end

-- Parse a single line containing a key-value pair (and possibly some comments)
-- Add the key-value pair to targetTable.
function parseKeyValueCfgLine(file, line, targetTable, allowAssignVar)
	
	-- Remove comments
	line = string.gsub(line, "#.*", ")
	line = string.gsub(line, "^//.*", ")         -- remove text with double slashes at the beginning
	line = string.gsub(line, "([^:])//.*", "%1")  -- remove text with double slashes unless preceeded by a colon
	line = string.gsub(line, "^%s*%-%-.*", ")    -- remove text with double hypens at the beginning, allowing for whitespaces for indented lines

	-- Fix nonbreaking space which is commonly inadvertently inserted into cfg files
	line = string.gsub(line, "\xc2\xa0", " ")		-- replace non-breaking space with regular space

	if string.find(line, "[\128-\255]") then
		error("Unsupported non-ascii characters found in " .. file .. " at line: " .. line)
	end

	-- Split into key, value pairs
	local key, value = string.match(line, "(%S+)%s+(.+)%s*")

	if key and value then
		-- Handle dotted key names

		local currentTable = targetTable
		local first, second
		second = key
		repeat
			first, second = string.match(second, "([^%.]+)%.?(.*)")

			if first and second then
				if second:len() == 0 then
					currentTable[first] = trim(value)
					if allowAssignVar then
						currentTable[first .. "_AllowAssignVar"] = true
					end
					if log then
						log:debug(string.format("set %s = %s (from key-value cfg %s)", key, value, file))
					end
				else
					if not currentTable[first] then
						currentTable[first] = {}
					end
					currentTable = currentTable[first]
				end
			end

		until not second or second:len() == 0
	end
end

-- Parse a string of key-value config entries
function parseKeyValueCfgString(location, str, targetTable, allowAssignVar)
	for line in string.gmatch(str, "[^\n]*") do
		parseKeyValueCfgLine(location, line, targetTable, allowAssignVar)
	end
end

-- Parse an old-style key-value CFG file
function readKeyValueCfg(file, targetTable, allowAssignVar, fileNotFoundIsError)
	if log then
		log:debug("Reading cfg '" .. file .. "'")
	end

	local f = vfs.open(file, "r")
	
	if not f then
		-- File not found
		if log then
			if fileNotFoundIsError then
				log:error("Failed to find file '" .. file .. "'")
			else
				log:debug("Failed to find file '" .. file .. "'")
			end
		end
			
		return
	end

	local firstLine = true

	while true do
		local line = f:read("*line")
		
		if not line then 
			break 
		end

		if firstLine and line == "#!/bin/lua" then
			executeCfg(f)
		end
		
		parseKeyValueCfgLine(file, line, targetTable, allowAssignVar)
	end
	
	f:close();
end
