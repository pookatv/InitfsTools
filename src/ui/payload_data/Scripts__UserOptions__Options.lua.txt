[description]
Core settings initialization script - handles level loading, quality presets, platform detection, and configuration file processing across all Frostbite games.

[version]
2011 (Battlefield 3 Alpha)
[notes]
Foundation version establishing core patterns.
Structure:
- Simple level name handling with Venice naming convention (Levels/name/name)
- Basic applySettings() for parsing config strings
- Loads 6 separate modules: Animations, Mesh, Physics, Shaders, Sound, Textures
- Reads /local/local.cfg for user overrides
- Processes command-line arguments directly (no conditional checks)
No Quality System:
- No applyQualitySettings() function yet
- No preset system for low/medium/high/ultra
- Direct module loading without platform detection
Command-Line Processing:
- Supports both table.member (-Graphics.Quality high) and simple variable (-level MP_001) syntax
- Always processes command-line, no allowCommandlineSettings gate
[content]
core = require "Frost.Core"
logging = require "Frost.Logging"


levelName = currentGameLevel or cmdLineOption('level') or Game.Level
if levelName then
	-- Level names without slashes are assumed to be shorthand for
	-- Levels/name/name to go with Venice's epic level naming convention
	if not string.find(levelName, "/") then
		levelName = "Levels/" .. levelName .. "/" .. levelName
	end

	Game.FastStartupEnable = true
	if not Game.Level then
		Game.Level = levelName
	end
end


function applySettings(settings)
	core.parseKeyValueCfgString(settings, _G)
end

dofile("Scripts/UserOptions/Options/Animations.lua")
dofile("Scripts/UserOptions/Options/Mesh.lua")
dofile("Scripts/UserOptions/Options/Physics.lua")
dofile("Scripts/UserOptions/Options/Shaders.lua")
dofile("Scripts/UserOptions/Options/Sound.lua")
dofile("Scripts/UserOptions/Options/Textures.lua")

if not cmdLineOption('noLocalCfg', false) then
	core.readKeyValueCfg("/local/local.cfg", _G)
end

for index,option in ipairs(commandLine) do
	local table,member = string.match(option, "^%-(%w+)%.(%w+)")

	if table and member then
		local value = commandLine[index+1] 
		if not _G[table] then _G[table] = {} end
		_G[table][member] = value 
	else
		local varName = string.match(option, "^%-([%w-]+)")
		if varName then
			local value = commandLine[index+1] 
			if value then
				_G[varName] = value
			end
		end
	end
end

function applyQualitySettings(name, qualityMap)
	local quality = settings[name]
	if quality == nil then
		log:error("Invalid setting '" .. name .. "'")
		return
	end
	
	local consoleCommands = qualityMap[quality]
	if consoleCommands == nil then
		log:error("Could not find console settings for '" .. name .. "' with quality " .. quality)
	else
		log:info("setting '" .. name .. "': " .. quality)
		applySettings(consoleCommands)
	end
end

[notes]
Sound.lua removed from Battlefield series.
Module Change:
- Only loads Graphics/GraphicsMobile, Physics, Animations
- Sound.lua completely removed
Why Remove Sound:
Battlefield's sound engine (Frostbite Audio) became fully data-driven
No Lua scripting needed for audio configuration
All sound settings moved to:
- AudioSettings (binary data files)
- Soundbank streaming (Wwise/FMOD integration)
- Console commands only (no quality presets)
Modern audio engines handle:
- Dynamic mixing (gunfire priority over ambience)
- 3D positional audio (HRTF, surround)
- Doppler effects, reverb zones
- All configured in visual tools, not scripts
This change persists through Battlefield V (2018)
